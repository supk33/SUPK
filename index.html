ทำตาม 4 ขั้นตอน แล้วระบบจะพร้อมใช้งานจริง
1
สร้าง Supabase Project
ไปที่ supabase.com → New Project → จดไว้: Project URL และ anon key
2
สร้าง Database Schema
ไปที่ SQL Editor ใน Supabase → วาง SQL นี้แล้วกด Run:
-- 1. ข้อมูลโปรไฟล์ผู้ใช้ create table profiles ( id uuid references auth.users primary key, name text, age int, sex text, weight numeric, height numeric, blood_type text, disease text, allergy text, doctor text, hospital text, created_at timestamptz default now(), updated_at timestamptz default now() ); -- 2. ผลตรวจสุขภาพแต่ละครั้ง create table health_records ( id uuid default gen_random_uuid() primary key, user_id uuid references auth.users not null, record_date date not null, hospital text, note text, item_count int, abnormal_count int, confidence numeric, raw_data jsonb, created_at timestamptz default now() ); -- 3. รายการผลตรวจแต่ละค่า create table health_items ( id uuid default gen_random_uuid() primary key, record_id uuid references health_records on delete cascade, user_id uuid references auth.users, name_th text, name_en text, value text, unit text, reference_range text, status text, confidence numeric, recorded_at date ); -- 4. เป้าหมายสุขภาพ create table health_goals ( id uuid default gen_random_uuid() primary key, user_id uuid references auth.users not null, goals text[], updated_at timestamptz default now() ); -- 5. ยาและอาหารเสริม create table medications ( id uuid default gen_random_uuid() primary key, user_id uuid references auth.users not null, name text, dose text, med_type text, active boolean default true, created_at timestamptz default now() ); -- Row Level Security (ผู้ใช้เห็นเฉพาะข้อมูลตัวเอง) alter table profiles enable row level security; alter table health_records enable row level security; alter table health_items enable row level security; alter table health_goals enable row level security; alter table medications enable row level security; -- Policies create policy "own data only" on profiles for all using (auth.uid() = id); create policy "own data only" on health_records for all using (auth.uid() = user_id); create policy "own data only" on health_items for all using (auth.uid() = user_id); create policy "own data only" on health_goals for all using (auth.uid() = user_id); create policy "own data only" on medications for all using (auth.uid() = user_id);
3
เปิด Google Auth
ไปที่ Authentication → Providers → Google → เปิด Enable
จากนั้นไปที่ Google Cloud Console → สร้าง OAuth 2.0 Client ID → นำ Client ID & Secret มาใส่ใน Supabase

เพิ่ม Redirect URL ใน Google Console:
https://[your-project].supabase.co/auth/v1/callback
4
ใส่ Supabase Credentials ด้านล่าง
คัดลอกจาก Settings → API ใน Supabase Dashboard
